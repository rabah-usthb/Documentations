\section{Queries}

\subsection{Employees That Have Salary Between 2000-4000 }

\begin{codeInput}{javascript}{Salary 2000-4000}{code01}
// first method implicit and 
db["Employee"].find({"Salaire" : {$gte : 2000 , $lte : 4000}} , {"Nom" : 1, "Poste" : 1, "Salaire" : 1, "_id": 0})

// second method explicit and
db["Employee"].find( { 
    $and : [
        {"Salaire" : {$gte : 2000} } , 
        {"Salaire" : {$lte : 4000} } 
    ]
    } , 
    {"Nom" : 1, "Poste" : 1, "Salaire" : 1, "_id": 0}
)
\end{codeInput}

\vspace{0.25cm}

\subsection{Engineer Employees That Have Salary Between 2000-4000}

\begin{codeInput}{javascript}{Enginner With Salary 2000-4000}{code01}
// first method implicit and 
db["Employee"].find({
    "Salaire" : {$gte : 2000 , $lte : 4000},
    "Poste" : /Ingenieur/i
    } , 
    {"Nom" : 1, "Salaire" : 1, "_id": 0}
)

// second method explicit and
db["Employee"].find( { 
    $and : [
        {"Salaire" : {$gte : 2000} } , 
        {"Salaire" : {$lte : 4000} } ,
        {"Post" : /Ingenieur/i}
    ]
    } , 
    {"Nom" : 1, "Salaire" : 1, "_id": 0}
)
\end{codeInput}

\subsection{Engineer Or Manager Employees}

\begin{codeInput}{javascript}{Enginner Or Manager}{code01}
// first method using in operator 
db["Employee"].find({
    "Poste" : {$in : [/Ingenieur/i, /Manageur/i]}
    },
    {"Nom" : 1, "Poste" : 1, "_id": 0}
)

// second method using or operator
db["Employee"].find( { 
    $or : [
        {"Post" : /Ingenieur/i},
        {"Post" : /Manageur/i}
    ]
    } , 
    {"Nom" : 1, "Poste" : 1, "_id": 0}
)
\end{codeInput}

\vspace{0.25cm}
